32  import com.jcabi.log.Logger;
33  import com.jcabi.xml.XML;
34  import com.jcabi.xml.XMLDocument;
35  import java.io.File;
36  import java.io.IOException;
37  import java.util.Collection;
38  import java.util.Map;
39  import java.util.concurrent.ConcurrentHashMap;
40  import javax.validation.constraints.NotNull;
41  import lombok.EqualsAndHashCode;
42  import lombok.ToString;
43  import org.apache.maven.plugin.AbstractMojo;
44  import org.apache.maven.plugin.MojoFailureException;
45  import org.apache.maven.plugins.annotations.LifecyclePhase;
46  import org.apache.maven.plugins.annotations.Mojo;
47  import org.apache.maven.plugins.annotations.Parameter;
48  import org.requs.Compiler;
49  import org.slf4j.impl.StaticLoggerBinder;
50  
51  /**
52   * Compile requs sources into XML.
53   *
54   * @author Yegor Bugayenko (yegor@tpc2.com)
55   * @version $Id$
56   * @since 1.1
57   * @checkstyle VisibilityModifierCheck (500 lines)
58   */
59  @ToString
60  @EqualsAndHashCode(callSuper = false)
61  @Mojo(
62      name = "compile", threadSafe = true,
63      defaultPhase = LifecyclePhase.COMPILE
64  )
65  public final class CompileMojo extends AbstractMojo {
66  
67      /**
68       * The source directory.
69       */
70      @Parameter(
71          required = true,
72          defaultValue = "${basedir}/src/main/requs"
73      )
74      @NotNull
75      public transient File input;
76  
77      /**
78       * Output directory.
79       */
80      @Parameter(
81          required = true,
82          defaultValue = "${project.build.directory}/requs"
83      )
84      @NotNull
85      public transient File output;
86  
87      /**
88       * Optional properties/options.
89       * @since 1.14
90       */
91      @Parameter(required = false)
92      @NotNull
93      @SuppressWarnings("PMD.UseConcurrentHashMap")
94      public transient Map<String, String> options =
95          new ConcurrentHashMap<String, String>(0);
96  
97      @Override
98      public void execute() throws MojoFailureException {
99          StaticLoggerBinder.getSingleton().setMavenLog(this.getLog());
100         if (!this.input.exists()) {
101             Logger.info(this, "input directory %s is absent", this.input);
102             return;
103         }
104         try {
105             new Compiler(this.input, this.output, this.options).compile();
106             final XML srs = new XMLDocument(new File(this.output, "requs.xml"));
107             final Collection<XML> errors = srs.nodes("//errors/error");
108             final int prefix = this.input.getAbsolutePath().length() + 1;
109             if (!errors.isEmpty()) {
110                 for (final XML error : errors) {
111                     Logger.error(
112                         this, "%s[%s:%s] %s",
113                         srs.xpath(
114                             String.format(
115                                 "/spec/files/file[@id='%d']/text()",
116                                 Integer.parseInt(error.xpath("@file").get(0))
117                             )
118                         ).get(0).substring(prefix),
119                         error.xpath("@line").get(0),
120                         error.xpath("@pos").get(0),
121                         error.xpath("text()").get(0)
122                     );
123                 }
124                 throw new MojoFailureException(
125                     String.format(
126                         "%d requs error(s), see log above",
127                         errors.size()
128                     )
129                 );
130             }
131         } catch (final IOException ex) {
132             throw new IllegalArgumentException("IO failure", ex);
133         }
134     }
